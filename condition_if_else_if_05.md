	LENGTH A 8 B 8 C 8;
	INPUT A B C;
	DATALINES;

RUN;

DATA TIPO;
	SET _3ANGULO;
	LENGTH TRIANGULO $32;

	IF ((A+B)<C) OR ((A+C)<B) OR ((B+C)<A) THEN
		TRIANGULO='No es un Triangulo';
	ELSE IF (A=B) AND (A=C) AND (B=C) THEN
		TRIANGULO='Equilatero';
	ELSE IF (A=B) OR (A=C) OR (B=C) THEN
		TRIANGULO='Isosceles';
	ELSE IF NOT(A=B) AND NOT(A=C) AND NOT(B=C) THEN
		TRIANGULO='Escaleno';
RUN;
