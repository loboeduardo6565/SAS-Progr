DATA INVERSOR;
	LENGTH NOMBRE $32 CapIni 8 PorcIntGen 8 AniosDinInv 8;
	INFILE DATALINES DLM=',';
	INPUT NOMBRE $ CapIni PorcIntGen AniosDinInv;
	DATALINES;

;
RUN;

DATA FINAL;
	SET INVERSOR;
	LENGTH MONTOPERCIBIDO $32;
TOTAL=CapIni;
	DO UNTIL(AniosDinInv=1);
		MONTOPERCIBIDO=((PorcIntGen/100)*TOTAL * (AniosDinInv));
		TOTAL=CapIni+MONTOPERCIBIDO;
		AniosDinInv=AniosDinInv-1;
		/*PUT AniosDinInv;*/
	END;
RUN;
